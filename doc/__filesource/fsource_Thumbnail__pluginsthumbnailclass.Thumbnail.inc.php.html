  232.     protected function readSourceImageData() {
  233.         if (!file_exists($this->image_path)) { // if source pic wasnt found
  234.             $this->image_path = 'error_pic';
  235.             $this->image_width =& $this->thumb_max_width;
  236.             $this->image_height =& $this->thumb_max_height;
  237.             $this->image = @imagecreatetruecolor($this->image_width, $this->image_height)
  238.                             or die ('Cannot Initialize new GD image stream');
  239.             $text_color = imagecolorallocate($this->image, 255, 255, 255);
  240.             imagestring($this->image, 1, 2,
  241.                         ($this->image_height / 2 - 10),
  242.                         "Could't find", $text_color);
  243.             imagestring($this->image, 1, 2,
  244.                         ($this->image_height / 2 - 4),
  245.                         'source image', $text_color);
  246.             imagestring($this->image, 1, 2,
  247.                         ($this->image_height / 2 + 4),
  248.                         '(Thumbnail V' . self::VERSION . ')', $text_color);
  249.         } else {
  250.             if (!isset($this->image_width)) {
  251.                 list($this->image_width, $this->image_height, $this->image_type, $attr) = getimagesize($this->image_path);
  252.                 unset($attr);
  253.                 if (!in_array($this->image_type, $this->formats)) {
  254.                     die("Can't create thumbnail from '" . $this->image_type . "' source: " . $this->image_path);
  255.                 } // end if
  256.             } // end if
  257.         } // end if
  258.     } // end function

  269.     protected function readSourceImage() {

  272.             switch ($this->image_type) {
  273.                 case 1:
  274.                     $this->image = imagecreatefromgif($this->image_path);
  275.                     break;
  276.                 case 2:
  277.                     $this->image = imagecreatefromjpeg($this->image_path);
  278.                     break;
  279.                 case 3:
  280.                     $this->image = imagecreatefrompng($this->image_path);
  281.                     break;
  282.                 case 15:
  283.                     $this->image = imagecreatefromwbmp($this->image_path);
  284.                     break;
  285.                 case 999:
  286.                 default:
  287.                     $this->image = imagecreatefromstring($this->image_path);
  288.                     break;
  289.             } // end switch
  290.         } // end if
  291.     } // end function

  293.     /**
  294.     * sets the actual width and height of the thumbnail based on the source image size and the max limits for the thumbnail
  295.     *
  296.     * @return void 
  297.     * @uses readSourceImageData()
  298.     * @uses $image_height
  299.     * @uses $thumb_max_height
  300.     * @uses $image_width
  301.     * @uses $thumb_max_width
  302.     */
  303.     protected function setThumbnailSize() {
  304.         $this->readSourceImageData();
  305.         if (($this->image_height > $this->thumb_max_height)
  306.             || ($this->image_width > $this->thumb_max_width)) {
  307.             $sizefactor = (double) (($this->image_height > $this->image_width) ? ($this->thumb_max_height / $this->image_height) : ($this->thumb_max_width / $this->image_width));
  308.         } else {
  309.             $sizefactor = (int) 1;
  310.         } // end if
  311.         $this->thumbnail_width = (int) ($this->image_width * $sizefactor);
  312.         $this->thumbnail_height = (int) ($this->image_height * $sizefactor);
  313.  
  314.         $sizefactor = (int) 1;
  315.         if ($this->image_height > $this->image_width) {
  316.             if ($this->thumbnail_width > $this->thumb_max_width) {
  317.                 $sizefactor = (double) ($this->thumb_max_width / $this->thumbnail_width);
  318.             } // end if
  319.         } else {
  320.             if ($this->thumbnail_height > $this->thumb_max_height) {
  321.                 $sizefactor = (double) ($this->thumb_max_height / $this->thumbnail_height);
  322.             } // end if
  323.         } // end if // end if
  324.         $this->thumbnail_width = (int) ($this->thumbnail_width * $sizefactor);
  325.         $this->thumbnail_height = (int) ($this->thumbnail_height * $sizefactor);
  326.         unset($sizefactor);
  327.     } // end function

  357.                                  $this->thumbnail_width, $this->thumbnail_height,
  358.                                  $this->image_width, $this->image_height);
  359.             } // end if
  360.         } // end if
  361.     } // end function

  391.             case 3:

  398.                 break;
  399.         } // end switch
  400.         imagedestroy($this->thumbnail);
