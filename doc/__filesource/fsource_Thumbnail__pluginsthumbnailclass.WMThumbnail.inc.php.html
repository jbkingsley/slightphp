  182.     protected function createThumbnail() {
  183.         parent::createThumbnail();
  184.         imagealphablending($this->thumbnail, true);
  185.         foreach ($this->logos as $logo) {
  186.             if (strlen(trim($logo['path'])) > 0) {
  187.                 $this->readWMImage($logo['path']);
  188.                 $start_pos_x = $this->thumbnail_width - $logo['margin'] - $this->wm_image_width;
  189.                 $start_pos_y = $this->thumbnail_height - $logo['margin'] - $this->wm_image_height;
  190.                 switch ($logo['pos']) {
  191.                     case 1: // left-top
  192.                         imagecopy($this->thumbnail, $this->wm_image,
  193.                                   $logo['margin'], $logo['margin'], 0, 0,
  194.                                   $this->wm_image_width,
  195.                                   $this->wm_image_height);
  196.                         break;
  197.                     case 2: // right-top
  198.                         imagecopy($this->thumbnail, $this->wm_image, $start_pos_x,
  199.                                   $logo['margin'], 0, 0, $this->wm_image_width,
  200.                                   $this->wm_image_height);
  201.                         break;
  202.                     case 3: // right-bottom
  203.                         imagecopy($this->thumbnail, $this->wm_image, $start_pos_x,
  204.                                   $start_pos_y, 0, 0, $this->wm_image_width,
  205.                                   $this->wm_image_height);
  206.                         break;
  207.                     case 4: // left-bottom
  208.                         imagecopy($this->thumbnail, $this->wm_image,
  209.                                   $logo['margin'], $start_pos_y, 0, 0,
  210.                                   $this->wm_image_width, $this->wm_image_height);
  211.                         break;
  212.                     case 5: // center
  213.                     default:
  214.                         $middle_x = ($this->thumbnail_width >> 1) - ($this->wm_image_width >> 1);
  215.                         $middle_y = ($this->thumbnail_height >> 1) - ($this->wm_image_height >> 1);
  216.                         imagecopy($this->thumbnail, $this->wm_image, $middle_x,
  217.                                   $middle_y, 0, 0, $this->wm_image_width,
  218.                                   $this->wm_image_height);
  219.                         break;
  220.                 } // end switch
  221.                 unset($this->wm_image);
  222.             } // end if
  223.         } // end foreach
