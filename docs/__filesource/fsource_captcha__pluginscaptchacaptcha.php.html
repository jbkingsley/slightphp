  256.     protected function GetRandomCaptchaText($length = null) {

  258.             $length = rand($this->minWordLength, $this->maxWordLength);
  259.         }
  260.

  263.  
  264.         $text  = "";
  265.         $vocal = rand(0, 1);
  266.         for ($i=0; $i<$length; $i++) {

  268.                 $text .= substr($vocals, mt_rand(0, 4), 1);
  269.             } else {
  270.                 $text .= substr($words, mt_rand(0, 22), 1);
  271.             }
  272.             $vocal = !$vocal;
  273.         }
  274.         return $text;
  275.     }

  286.      * Random dictionary word generation

  288.      * @param boolean $extended Add extended "fake" words
  289.      * @return string Word

  291.     function GetDictionaryCaptchaText($extended = false) {
  292.         if (empty($this->wordsFile)) {
  293.             return false;
  294.         }
  295.  
  296.         $fp     = fopen($this->wordsFile, "r");
  297.         $length = strlen(fgets($fp));
  298.         if (!$length) {
  299.             return false;
  300.         }
  301.         $line   = rand(0, (filesize($this->wordsFile)/$length)-1);
  302.         if (fseek($fp, $length*$line) == -1) {
  303.             return false;
  304.         }
  305.         $text = trim(fgets($fp));
  306.         fclose($fp);
  307.  
  308.  
  309.         /** Change ramdom volcals */
  310.         if ($extended) {
  311.             $text   = str_split($text, 1);
  312.             $vocals = array('a', 'e', 'i', 'o', 'u');
  313.             foreach ($text as $i => $char) {
  314.                 if (mt_rand(0, 1) && in_array($char, $vocals)) {
  315.                     $text[$i] = $vocals[mt_rand(0, 4)];
  316.                 }
  317.             }
  318.             $text = implode('', $text);
  319.         }
  320.  
  321.         return $text;
  322.     }

  336.     protected function WriteText($text, $fontcfg = array()) {
  337.         if (empty($fontcfg)) {
  338.             // Select the font configuration
  339.             $fontcfg  = $this->fonts[array_rand($this->fonts)];
  340.         }
  341.         $fontfile = dirname(__FILE__).'/fonts/'.$fontcfg['font'];
  342.  
  343.         /** Increase font-size for shortest words: 9% for each glyp missing */
  344.         $lettersMissing = $this->maxWordLength-strlen($text);
  345.         $fontSizefactor = 1+($lettersMissing*0.09);
  346.  
  347.         // Text generation (char by char)
  348.         $x      = 20*$this->scale;
  349.         $y      = round(($this->height*27/40)*$this->scale);
  350.         $length = strlen($text);
  351.         for ($i=0; $i<$length; $i++) {
  352.             $degree   = rand($this->maxRotation*-1, $this->maxRotation);
  353.             $fontsize = rand($fontcfg['minSize'], $fontcfg['maxSize'])*$this->scale*$fontSizefactor;
  354.             $letter   = substr($text, $i, 1);
  355.  
  356.             if ($this->shadowColor) {
  357.                 $coords = imagettftext($this->im, $fontsize, $degree,
  358.                     $x+$this->scale, $y+$this->scale,
  359.                     $this->GdShadowColor, $fontfile, $letter);
  360.             }
  361.             $coords = imagettftext($this->im, $fontsize, $degree,
  362.                 $x, $y,
  363.                 $this->GdFgColor, $fontfile, $letter);
  364.             $x += ($coords[2]-$x) + ($fontcfg['spacing']*$this->scale);
  365.         }
  366.     }

  373.     protected function WaveImage() {
  374.         // X-axis wave generation
  375.         $xp = $this->scale*$this->Xperiod*rand(1,3);
  376.         $k = rand(0, 100);
  377.         for ($i = 0; $i < ($this->width*$this->scale); $i++) {
  378.             imagecopy($this->im, $this->im,
  379.                 $i-1, sin($k+$i/$xp) * ($this->scale*$this->Xamplitude),
  380.                 $i, 0, 1, $this->height*$this->scale);
  381.         }
  382.  
  383.         // Y-axis wave generation
  384.         $k = rand(0, 100);
  385.         $yp = $this->scale*$this->Yperiod*rand(1,2);
  386.         for ($i = 0; $i < ($this->height*$this->scale); $i++) {
  387.             imagecopy($this->im, $this->im,
  388.                 sin($k+$i/$yp) * ($this->scale*$this->Yamplitude), $i-1,
  389.                 0, $i, $this->width*$this->scale, 1);
  390.         }
  391.     }
